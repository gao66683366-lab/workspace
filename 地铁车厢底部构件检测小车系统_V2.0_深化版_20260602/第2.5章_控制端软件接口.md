## 2.5.3 控制指令接口（TCP:5000）

### 2.5.3.1 帧格式定义

```c
// 统一指令帧结构
struct RobotCommand {
    uint16_t header;      // 帧头 0xAA55
    uint16_t length;      // Payload长度
    uint8_t  cmd;         // 主命令字
    uint8_t  subcmd;      // 子命令字
    uint8_t  payload[];   // 参数数据
    uint16_t checksum;    // CRC16校验
};
```

### 2.5.3.2 命令字定义

| cmd | subcmd | 功能 | Payload | 说明 |
|---|---|---|---|---|
| 0x01 | 0x01 | 设置速度 | `float speed` | speed: 0.0~0.5 m/s |
| 0x01 | 0x02 | 前进 | - | 开始前进 |
| 0x01 | 0x03 | 后退 | - | 开始后退 |
| 0x01 | 0x04 | 停止 | - | 立即停止 |
| 0x01 | 0x05 | 急停 | - | 硬件急停，最快响应 |
| 0x02 | 0x01 | 云台角度 | `uint8 cam; int16 pan; int16 tilt` | cam:0前/1后, pan/tilt: 度 |
| 0x02 | 0x02 | 云台复位 | `uint8 cam` | 回到中立位 |
| 0x02 | 0x03 | 预设位保存 | `uint8 cam; uint8 preset_id` | 保存当前角度为预设位 |
| 0x02 | 0x04 | 预设位调用 | `uint8 cam; uint8 preset_id` | 调用预设位 |
| 0x03 | 0x01 | 设置光源亮度 | `uint8 brightness` | 0~100% |
| 0x03 | 0x02 | 灯光开关 | `uint8 onoff` | 0:关 1:开 |
| 0x04 | 0x01 | 开始采集 | `uint8 mode` | mode:0自动/1手动 |
| 0x04 | 0x02 | 停止采集 | - | 停止本次采集 |
| 0x05 | 0x01 | 系统自检 | - | 请求系统自检 |
| 0x05 | 0x02 | 零点标定 | `uint8 sensor_type` | sensor_type:0里程/1姿态 |
| 0xFF | 0x01 | 心跳 | `uint32 timestamp` | 客户端存活确认 |

### 2.5.3.3 指令示例

```python
# Python 示例：发送速度设置指令
import socket
import struct
import json

class RobotClient:
    def __init__(self, ip, port=5000):
        self.sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        self.sock.connect((ip, port))
        self.sock.settimeout(5.0)

    def send_command(self, cmd, subcmd, payload=b''):
        header = 0xAA55
        length = 2 + len(payload)  # cmd + subcmd + payload
        frame = struct.pack('>HH', header, length)
        frame += bytes([cmd, subcmd])
        frame += payload
        checksum = self._crc16(frame)
        frame += struct.pack('>H', checksum)
        self.sock.sendall(frame)

    def set_speed(self, speed):
        payload = struct.pack('>f', speed)  # float32
        self.send_command(0x01, 0x01, payload)

    def move_forward(self):
        self.send_command(0x01, 0x02)

    def move_stop(self):
        self.send_command(0x01, 0x04)

    def ptz_control(self, camera, pan, tilt):
        payload = struct.pack('>Bhh', camera, int(pan), int(tilt))
        self.send_command(0x02, 0x01, payload)

    def set_light(self, brightness):
        self.send_command(0x03, 0x01, bytes([brightness]))

    def heartbeat(self):
        import time
        ts = int(time.time())
        payload = struct.pack('>I', ts)
        self.send_command(0xFF, 0x01, payload)

    @staticmethod
    def _crc16(data):
        crc = 0xFFFF
        for b in data:
            crc ^= b
            for _ in range(8):
                if crc & 0x0001:
                    crc = (crc >> 1) ^ 0xA001
                else:
                    crc >>= 1
        return crc
```

```csharp
// C# 示例：指令封装类
using System;
using System.Net.Sockets;
using System.Threading;

namespace RobotControlSDK
{
    public class RobotCommander : IDisposable
    {
        private TcpClient _client;
        private NetworkStream _stream;
        private readonly object _lock = new object();

        public RobotCommander(string ip, int port = 5000)
        {
            _client = new TcpClient(ip, port);
            _stream = _client.GetStream();
            _stream.ReadTimeout = 5000;
        }

        public void SendCommand(byte cmd, byte subcmd, byte[] payload = null)
        {
            payload ??= Array.Empty<byte>();
            byte[] frame = new byte[6 + payload.Length];
            // Header
            frame[0] = 0xAA;
            frame[1] = 0x55;
            // Length
            frame[2] = (byte)((2 + payload.Length) >> 8);
            frame[3] = (byte)(2 + payload.Length);
            // CMD
            frame[4] = cmd;
            frame[5] = subcmd;
            // Payload
            Array.Copy(payload, 0, frame, 6, payload.Length);
            // CRC16
            ushort crc = CalcCRC16(frame, frame.Length - 2);
            frame[frame.Length - 2] = (byte)(crc >> 8);
            frame[frame.Length - 1] = (byte)(crc & 0xFF);

            lock (_lock)
            {
                _stream.Write(frame, 0, frame.Length);
            }
        }

        // 速度设置 0.0~0.5 m/s
        public void SetSpeed(float speed)
        {
            byte[] payload = BitConverter.GetBytes(speed);
            SendCommand(0x01, 0x01, payload);
        }

        // 运动控制
        public void MoveForward() => SendCommand(0x01, 0x02);
        public void MoveBackward() => SendCommand(0x01, 0x03);
        public void MoveStop() => SendCommand(0x01, 0x04);
        public void MoveEstop() => SendCommand(0x01, 0x05);

        // 云台控制 camera:0前/1后
        public void PTZControl(byte camera, short pan, short tilt)
        {
            byte[] payload = new byte[] { camera,
                (byte)(pan >> 8), (byte)(pan & 0xFF),
                (byte)(tilt >> 8), (byte)(tilt & 0xFF) };
            SendCommand(0x02, 0x01, payload);
        }

        // 光源控制
        public void SetLightBrightness(byte brightness)
            => SendCommand(0x03, 0x01, new byte[] { brightness });

        // 心跳
        public void Heartbeat()
        {
            byte[] ts = BitConverter.GetBytes(DateTimeOffset.Now.ToUnixTimeSeconds());
            SendCommand(0xFF, 0x01, ts);
        }

        private static ushort CalcCRC16(byte[] data, int len)
        {
            ushort crc = 0xFFFF;
            for (int i = 0; i < len; i++)
            {
                crc ^= data[i];
                for (int j = 0; j < 8; j++)
                {
                    crc = (ushort)((crc & 1) != 0 ? (crc >> 1) ^ 0xA001 : crc >> 1);
                }
            }
            return crc;
        }

        public void Dispose()
        {
            _stream?.Dispose();
            _client?.Dispose();
        }
    }
}
```

## 2.5.4 传感器数据接口（UDP:5001）

### 2.5.4.1 数据帧结构

```c
// 统一数据帧头
struct SensorFrame {
    uint8_t  type;       // 数据类型
    uint8_t  length;     // 后续长度
    uint32_t timestamp;  // Unix时间戳（秒）
    uint32_t timestamp_us; // 微秒部分
    uint8_t  payload[];  // 数据负载
};
```

### 2.5.4.2 数据类型定义

| type | 数据类型 | 说明 | 更新频率 |
|---|---|---|---|
| 0x01 | IMU_DATA | 姿态传感器数据 | 50Hz |
| 0x02 | RANGING_DATA | 测距传感器数据 | 20Hz |
| 0x03 | ODOMETRY_DATA | 里程计数据 | 50Hz |
| 0x04 | BMS_DATA | BMS电池管理数据 | 1Hz |
| 0x05 | LIDAR_DATA | 激光雷达点云 | 10Hz |
| 0x06 | SYSTEM_STATUS | 系统状态 | 1Hz |
| 0xFF | HEARTBEAT_ACK | 心跳应答 | 1Hz |

### 2.5.4.3 数据字段定义

```c
// IMU数据 (type=0x01, length=36)
struct IMU_DATA {
    float roll;     // 横滚角，度，范围-180~+180
    float pitch;   // 俯仰角，度，范围-90~+90
    float yaw;     // 航向角，度，范围0~360
    float wx;      // 角速度X，deg/s
    float wy;      // 角速度Y，deg/s
    float wz;      // 角速度Z，deg/s
    float ax;      // 加速度X，g
    float ay;      // 加速度Y，g
    float az;      // 加速度Z，g
};

// 测距数据 (type=0x02, length=12)
struct RANGING_DATA {
    float front;   // 前方距离，mm
    float rear;    // 后方距离，mm
};

// 里程数据 (type=0x03, length=12)
struct ODOMETRY_DATA {
    float distance;  // 累计里程，mm
    float speed;    // 当前速度，m/s
};

// BMS数据 (type=0x04, length=28)
struct BMS_DATA {
    float voltage;     // 总电压，V
    float current;     // 电流，A（正=充电/负=放电）
    uint8_t soc;       // 荷电状态，%
    int8_t temp_max;   // 最高温度，℃
    int8_t temp_min;   // 最低温度，℃
    uint16_t cycle_count; // 循环次数
    uint8_t status;    // 状态位 0=正常/1=告警/2=故障
};

// 系统状态 (type=0x06, length=8)
struct SYSTEM_STATUS {
    uint8_t mode;        // 运行模式 0=待机/1=自动/2=手动/3=应急
    uint8_t error_code;  // 错误码 0=正常
    uint16_t cpu_temp;   // CPU温度，0.1℃
    uint16_t memory_usage; // 内存使用率，0.1%
};
```

### 2.5.4.4 数据接收示例

```python
# Python UDP数据接收
import socket
import struct
import threading
import json

class SensorDataReceiver:
    def __init__(self, port=5001):
        self.sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
        self.sock.bind(('', port))
        self.sock.settimeout(1.0)
        self.running = False
        self.callbacks = {}
        self.latest = {
            'imu': None,
            'ranging': None,
            'odometry': None,
            'bms': None,
            'lidar': None,
            'system': None
        }

    def start(self):
        self.running = True
        self.thread = threading.Thread(target=self._receive_loop)
        self.thread.daemon = True
        self.thread.start()

    def stop(self):
        self.running = False
        self.sock.close()

    def register_callback(self, data_type, callback):
        """注册数据回调
        data_type: 'imu'/'ranging'/'odometry'/'bms'/'lidar'/'system'
        callback: function(data_dict)
        """
        self.callbacks[data_type] = callback

    def _receive_loop(self):
        while self.running:
            try:
                data, addr = self.sock.recvfrom(4096)
                self._parse_frame(data)
            except socket.timeout:
                continue

    def _parse_frame(self, data):
        if len(data) < 8:
            return

        msg_type, length, ts, ts_us = struct.unpack('>BBII', data[:8])
        payload = data[8:8+length]

        if msg_type == 0x01:  # IMU
            if len(payload) >= 36:
                roll, pitch, yaw, wx, wy, wz, ax, ay, az = struct.unpack('>9f', payload[:36])
                d = {'roll': roll, 'pitch': pitch, 'yaw': yaw,
                     'wx': wx, 'wy': wy, 'wz': wz,
                     'ax': ax, 'ay': ay, 'az': az,
                     'timestamp': ts + ts_us/1000000.0}
                self.latest['imu'] = d
                self._emit('imu', d)

        elif msg_type == 0x02:  # RANGING
            if len(payload) >= 8:
                front, rear = struct.unpack('>ff', payload[:8])
                d = {'front': front, 'rear': rear,
                     'timestamp': ts + ts_us/1000000.0}
                self.latest['ranging'] = d
                self._emit('ranging', d)

        elif msg_type == 0x03:  # ODOMETRY
            if len(payload) >= 8:
                dist, speed = struct.unpack('>ff', payload[:8])
                d = {'distance': dist, 'speed': speed,
                     'timestamp': ts + ts_us/1000000.0}
                self.latest['odometry'] = d
                self._emit('odometry', d)

        elif msg_type == 0x04:  # BMS
            if len(payload) >= 20:
                vol, cur, soc, tmax, tmin, cycles, status = struct.unpack('>ffBbbBH', payload[:20])
                d = {'voltage': vol, 'current': cur, 'soc': soc,
                     'temp_max': tmax, 'temp_min': tmin,
                     'cycle_count': cycles, 'status': status,
                     'timestamp': ts + ts_us/1000000.0}
                self.latest['bms'] = d
                self._emit('bms', d)

        elif msg_type == 0x06:  # SYSTEM
            if len(payload) >= 4:
                mode, err, cpu_temp, mem = struct.unpack('>BBHH', payload[:4])
                d = {'mode': mode, 'error_code': err,
                     'cpu_temp': cpu_temp/10.0, 'memory_usage': mem/10.0,
                     'timestamp': ts + ts_us/1000000.0}
                self.latest['system'] = d
                self._emit('system', d)

    def _emit(self, key, data):
        if key in self.callbacks:
            self.callbacks[key](data)

# 使用示例
receiver = SensorDataReceiver(5001)

def on_imu(data):
    print(f"横滚:{data['roll']:.2f}° 俯仰:{data['pitch']:.2f}° 航向:{data['yaw']:.2f}°")

def on_bms(data):
    print(f"电压:{data['voltage']:.1f}V SOC:{data['soc']}%")

receiver.register_callback('imu', on_imu)
receiver.register_callback('bms', on_bms)
receiver.start()
```

### 2.5.5.2 视频解码示例

```python
# Python OpenCV视频接收示例
import cv2

class VideoReceiver:
    def __init__(self, front_url, rear_url):
        self.front_url = front_url
        self.rear_url = rear_url
        self.front_cap = None
        self.rear_cap = None

    def start(self):
        # RTSP认证（可选）
        # self.front_url = self.front_url.replace('rtsp://', 'rtsp://user:pass@')
        self.front_cap = cv2.VideoCapture(self.front_url)
        self.rear_cap = cv2.VideoCapture(self.rear_url)

    def get_frames(self):
        """返回(front_frame, rear_frame)，若无帧则返回(None, None)"""
        ok1, f = self.front_cap.read()
        ok2, r = self.rear_cap.read()
        return (f if ok1 else None, r if ok2 else None)

    def stop(self):
        if self.front_cap:
            self.front_cap.release()
        if self.rear_cap:
            self.rear_cap.release()

# 使用
receiver = VideoReceiver(
    'rtsp://192.168.1.100:8554/front',
    'rtsp://192.168.1.100:8554/rear'
)
receiver.start()

while True:
    front_frame, rear_frame = receiver.get_frames()
    if front_frame is not None:
        cv2.imshow('Front', front_frame)
    if rear_frame is not None:
        cv2.imshow('Rear', rear_frame)
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

receiver.stop()
cv2.destroyAllWindows()
```

## 2.5.7 错误码定义

| 错误码 | 名称 | 说明 | 处理建议 |
|---|---|---|---|
| 0x00 | ERR_OK | 正常 | - |
| 0x01 | ERR_TIMEOUT | 通信超时 | 检查WiFi连接，重试 |
| 0x02 | ERR_FRAME | 帧格式错误 | 固件版本不匹配，升级 |
| 0x03 | ERR_CRC | 校验失败 | 重发指令 |
| 0x10 | ERR_MOTOR_FAULT | 电机故障 | 检查电机连接 |
| 0x11 | ERR_OVERCURRENT | 过流保护 | 降低负载，减小速度 |
| 0x12 | ERR_OVERHEAT | 过温保护 | 停机冷却 |
| 0x20 | ERR_CAMERA_OFFLINE | 相机离线 | 检查网线连接 |
| 0x21 | ERR_CAMERA_TRIGGER | 相机触发失败 | 重启相机 |
| 0x30 | ERR_BMS_UNDERVOLT | 电池欠压 | 尽快充电 |
| 0x31 | ERR_BMS_OVERCURRENT | 电池过流 | 减小负载 |
| 0x32 | ERR_BMS_OVERTEMP | 电池过温 | 停机冷却 |
| 0x40 | ERR_SENSOR_OFFLINE | 传感器离线 | 检查485连接 |

## 2.5.9 章节小结

本章详细定义了控制端软件与机器人主控板之间的完整软件接口规范：

- **通信矩阵**：TCP控制指令(5000) / UDP传感器数据(5001) / RTSP视频流(8554)
- **控制指令协议**：统一帧格式(0xAA55帧头+CRC16)、命令字定义(运动/云台/光源/采集/系统)
- **传感器数据协议**：IMU/测距/里程/BMS/激光雷达/系统状态六类数据帧
- **视频流接口**：RTSP双通道(前视/后视) 1080P@30fps
- **SDK封装**：Python/C#完整示例代码
- **错误码体系**：通信错误/电机故障/相机故障/BMS故障/传感器故障

以上接口规范可用于控制软件RobotSDK的开发和移植。
